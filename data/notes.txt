RUN SVGO (Manual):

- $ npm install -g svgo
- $ svgo --input img/graph.svg --output img/optimised-graph.svg
- if it fucks up complex svgs, try: $ svgo --disable=convertPathData --disable=mergePaths -i img/a.svg

svgmin: {

    },
